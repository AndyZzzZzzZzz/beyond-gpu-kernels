\section{Motivation \& Background}

\begin{frame}{The Shift: From Chatbot to Autonomous Agent}
  \begin{columns}
    \begin{column}{0.58\textwidth}
      \begin{itemize}
        \item LLMs are rapidly evolving from \textbf{static chat interfaces} into \textbf{autonomous agents}.
        \item Modern agents \textit{reason}, \textit{call tools}, \textit{parse results}, and \textit{iterate}.
        \item Historical optimization focus: \textcolor{neonamber}{\textbf{GPU forward-pass}}
        \begin{itemize}
          \item Time-To-First-Token (TTFT)
          \item Inter-token latency
        \end{itemize}
        \item But agents spend most of their time \textbf{outside} the GPU\dots
      \end{itemize}
    \end{column}
    \begin{column}{0.42\textwidth}
      \begin{block}{Key Insight}
        The execution bottleneck is shifting \textbf{from the accelerator back to the host CPU}.
      \end{block}
      \vspace{0.3em}
      \begin{alertblock}{The Gap}
        Current literature treats the tool-execution pipeline as a \textit{black box}.
      \end{alertblock}
    \end{column}
  \end{columns}
\end{frame}

\begin{frame}{What is ``Tool Calling''? -- The ReAct Framework}
  \begin{columns}
    \begin{column}{0.55\textwidth}
      \textbf{ReAct = Reasoning + Acting}
      \begin{enumerate}
        \item LLM reads the user prompt
        \item LLM emits a \textbf{structured JSON} tool call
        \item Python orchestrator \textbf{parses} the JSON
        \item Orchestrator \textbf{executes} a native subprocess
        \item \texttt{stdout} is \textbf{injected back} into context
        \item Repeat until the task is complete
      \end{enumerate}
      \vspace{0.4em}
      \textcolor{neonpink}{\textbf{This loop is strictly serial.}}
    \end{column}
    \begin{column}{0.45\textwidth}
      \begin{tikzpicture}[node distance=0.55cm, font=\scriptsize]
        \tikzstyle{box}=[draw=neonmint, thick, rounded corners=1pt, fill=slate,
                          minimum width=3.3cm, minimum height=0.6cm, text centered, text=bone]
        \tikzstyle{arrow}=[->, thick, neonmint]
        \node[box] (llm) {LLM (GPU): emit JSON};
        \node[box, below=of llm] (parse) {Python: parse \& validate};
        \node[box, below=of parse] (exec) {C++ tool: execute};
        \node[box, below=of exec] (inject) {Python: inject result};
        \draw[arrow] (llm) -- (parse);
        \draw[arrow] (parse) -- (exec);
        \draw[arrow] (exec) -- (inject);
        \draw[arrow, neonpink] (inject.east) .. controls +(1.5,1) and +(1.5,-1) .. (llm.east)
              node[midway, right, xshift=2pt, text=neonpink] {loop};
      \end{tikzpicture}
    \end{column}
  \end{columns}
\end{frame}

\begin{frame}{Architectural Divergence: ML vs. Datacenter Workloads}
  \begin{columns}
    \begin{column}{0.5\textwidth}
      \begin{block}{GPU-Friendly (Matrix Math)}
        \begin{itemize}
          \item Dense matrix multiplication
          \item Massively parallel
          \item Predictable memory access
          \item High arithmetic intensity
        \end{itemize}
      \end{block}
    \end{column}
    \begin{column}{0.5\textwidth}
      \begin{alertblock}{Agentic Tool Workloads}
        \begin{itemize}
          \item JSON parsing, DB lookups, FS walks
          \item Serial, branch-heavy
          \item Pointer chasing $\rightarrow$ LLC misses
          \item Behave like \textbf{datacenter RPCs}
        \end{itemize}
      \end{alertblock}
    \end{column}
  \end{columns}
  \vspace{1em}
  \begin{center}
    \textcolor{neonmint}{\textbf{Agent tools are fundamentally hostile to GPU architectures.}}
  \end{center}
\end{frame}

\begin{frame}{Research Questions \& Contributions}
  \begin{exampleblock}{Research Questions}
    \begin{enumerate}
      \item How much CPU overhead does the LLM orchestration layer actually impose?
      \item Do classical hardware optimizations (SIMD, prefetching, OpenMP) help agentic tools?
      \item Does kernel-level efficiency propagate to end-to-end latency?
    \end{enumerate}
  \end{exampleblock}

  \textbf{Our three contributions:}
  \begin{enumerate}
    \item \textcolor{neonmint}{\textbf{Quantification of the Agentic Tax:}} up to $\mathbf{69.5\times}$ slowdown on lightweight tools.
    \item \textcolor{neonviolet}{\textbf{Microarchitectural characterization}} of compute-, memory-, and I/O-bound agent tools.
    \item \textcolor{neonpink}{\textbf{Identification of the Masking Effect:}} kernel-level wins rendered statistically invisible at the system level.
  \end{enumerate}
\end{frame}

\section{Methodology}

\begin{frame}{Decoupled System Architecture}
  \textbf{Goal:} Isolate microarchitectural overhead \textit{without} cross-contamination between Python and C++.
  \vspace{0.5em}
  \begin{columns}
    \begin{column}{0.58\textwidth}
      \begin{block}{Three Decoupled Components}
        \begin{enumerate}
          \item \textbf{Orchestrator (Python / FastAPI)}: prompt assembly, inference, JSON routing.
          \item \textbf{Execution Kernels (C++)}: compiled native binaries for compute / memory / I/O.
          \item \textbf{Static Workloads}: pre-generated deterministic stress payloads.
        \end{enumerate}
      \end{block}
    \end{column}
    \begin{column}{0.42\textwidth}
      \begin{alertblock}{Why Decouple?}
        \begin{itemize}
          \item Isolates C++ from Python GIL
          \item Avoids Python memory-management noise
          \item Enables pristine profiling
        \end{itemize}
      \end{alertblock}
    \end{column}
  \end{columns}
  \vspace{0.5em}
  \begin{center}
    \textcolor{mist}{\textbf{Model:} Qwen2.5-7B-Instruct (FP16 + SDPA), \texttt{do\_sample=False}}
  \end{center}
\end{frame}

\begin{frame}{PAPI: Hardware Performance Counters}
  \begin{columns}
    \begin{column}{0.5\textwidth}
      \textbf{Performance API} $\rightarrow$ direct access to the Linux kernel's Performance Monitoring Unit (PMU).
      \vspace{0.4em}
      \begin{block}{Four hardware events tracked}
        \begin{itemize}
          \item \texttt{PAPI\_TOT\_CYC} -- total cycles
          \item \texttt{PAPI\_TOT\_INS} -- total instructions
          \item \texttt{PAPI\_BR\_MSP} -- branch mispredictions
          \item \texttt{PAPI\_L3\_TCM} -- L3 cache misses
        \end{itemize}
      \end{block}
    \end{column}
    \begin{column}{0.5\textwidth}
      \textbf{Windowing strategy:}
      \begin{itemize}
        \item \textcolor{neonmint}{\textbf{CPU Window 1:}} prompt assembly + tokenization
        \item \textcolor{neonviolet}{\textbf{CPU Window 2:}} decoding + JSON extraction
      \end{itemize}
      \vspace{0.4em}
      \begin{exampleblock}{Testbed}
        Intel Xeon Gold 6242R @ 3.10 GHz -- isolated, predictable cycle-to-latency.
      \end{exampleblock}
      \small Hardware EventSets guarded in \texttt{try\dots finally} to avoid PMU starvation.
    \end{column}
  \end{columns}
\end{frame}

\begin{frame}{Three C++ Tool Kernels}
  \begin{columns}
    \begin{column}{0.33\textwidth}
      \begin{block}{Compute-Bound}
        \textbf{AST Math Evaluator}
        \begin{itemize}
          \item Nested math expressions
          \item Recursive parser
          \item Stresses \textbf{branch prediction}
          \item Tests instruction throughput
        \end{itemize}
      \end{block}
    \end{column}
    \begin{column}{0.33\textwidth}
      \begin{block}{Memory-Bound}
        \textbf{SQLite DB Scan}
        \begin{itemize}
          \item Forced full-table scan
          \item \texttt{LIKE} bypasses indexes
          \item Thrashes \textbf{LLC (L3)}
          \item Classic memory wall
        \end{itemize}
      \end{block}
    \end{column}
    \begin{column}{0.33\textwidth}
      \begin{block}{I/O-Bound}
        \textbf{Directory Walker}
        \begin{itemize}
          \item Recursive FS traversal
          \item OS metadata queries
          \item Stresses \textbf{syscalls}
          \item Context-switch heavy
        \end{itemize}
      \end{block}
    \end{column}
  \end{columns}
  \vspace{0.8em}
  \begin{center}
    \textcolor{neonmint}{\textbf{Each kernel manages its own PAPI profiling around the core workload.}}
  \end{center}
\end{frame}

\begin{frame}{The Control Baseline: \texttt{pipeline\_mode="direct"}}
  \begin{columns}
    \begin{column}{0.55\textwidth}
      \textbf{Question:} How do we prove the ``Agentic Tax'' exists?
      \vspace{0.5em}
      \begin{block}{Direct Baseline}
        \begin{itemize}
          \item GPU and LLM \textbf{completely bypassed}
          \item Regex extracts target paths from prompt
          \item C++ kernels invoked directly
          \item Represents the theoretical ``speed of light''
        \end{itemize}
      \end{block}
      \vspace{0.3em}
      \textbf{Result:} Subtracting Direct from Full Agentic gives us the \textbf{exact cost of orchestration}.
    \end{column}
    \begin{column}{0.45\textwidth}
      \begin{alertblock}{Full Agentic Loop}
        Qwen-2.5-7B autonomously:
        \begin{itemize}
          \item Parses prompt
          \item Emits JSON
          \item Executes tool
          \item Re-injects result
        \end{itemize}
      \end{alertblock}
      \begin{center}
        \textcolor{neonmint}{\textbf{Direct vs. Agentic}}\\
        \small\textcolor{mist}{= the Agentic Tax}
      \end{center}
    \end{column}
  \end{columns}
\end{frame}

\begin{frame}{Hardware Optimizations Applied}
  \begin{columns}
    \begin{column}{0.5\textwidth}
      \begin{block}{1. SIMD Vectorization (AVX2)}
        \small Applied to the AST math evaluator. 256-bit wide SIMD registers intended to accelerate numerical parsing.
      \end{block}
      \begin{block}{2. Software Prefetching}
        \small \texttt{\_\_builtin\_prefetch} with lookahead distance of 32 elements. Latency hiding for cold-cache DB scans.
      \end{block}
    \end{column}
    \begin{column}{0.5\textwidth}
      \begin{block}{3. OpenMP Concurrency}
        \small Task-parallel directory walker. Overlaps I/O-bound system calls across threads.
      \end{block}
      \vspace{0.3em}
      \begin{exampleblock}{Compiler Flags}
        \texttt{-O3 -march=native -mavx2} \\
        \texttt{-O3 -march=native -fopenmp}
      \end{exampleblock}
    \end{column}
  \end{columns}
  \vspace{0.6em}
  \begin{center}
    \textcolor{neonamber}{\textbf{Do these optimizations matter in an agentic context?}}
  \end{center}
\end{frame}

\begin{frame}{The Agentic Tax is a Fixed-Cost Penalty}
  \begin{columns}
    \begin{column}{0.55\textwidth}
      \begin{itemize}
        \item \textcolor{neonpink}{\textbf{I/O walker:} $69.5\times$ slowdown}\\
              \small Native: $\sim$15 ms $\rightarrow$ Agentic: $\sim$1 s
        \item \textcolor{neonamber}{\textbf{AST math:} $14.9\times$ slowdown}
        \item \textcolor{neonice}{\textbf{DB scan:} only $2.4\times$ slowdown}\\
              \small $\sim$764 ms native, Agentic tax amortized
      \end{itemize}
      \vspace{0.5em}
      \begin{alertblock}{Key Finding}
        The tax is \textbf{not} proportional; it's a \textbf{massive fixed cost} that disproportionately punishes lightweight tools.
      \end{alertblock}
    \end{column}
    \begin{column}{0.45\textwidth}
      \begin{block}{Amdahl Saves the DB scan}
        The heavy 2.5 billion-cycle DB workload is so long that the orchestration overhead becomes a \textit{proportionally} smaller slice.
      \end{block}
      \vspace{0.4em}
      \begin{center}
        \textcolor{neonmint}{\textbf{For most tools, the framework dominates execution time.}}
      \end{center}
    \end{column}
  \end{columns}
\end{frame}

\begin{frame}{Two Phenomena Revealed by Multi-Step Telemetry}
  \begin{columns}
    \begin{column}{0.5\textwidth}
      \begin{block}{1. Framework: Cold Start + Drift}
        \begin{itemize}
          \item \textbf{Step 1:} $\sim$82M instructions\\
                \small one-time JSON schema load
          \item \textbf{Steps 2--4:} baseline $\sim$13.7M\\
                \small grows linearly to 18.8M
          \item Cause: \textbf{stateless re-serialization} of the growing context window
          \item Framework IPC rises 1.65 $\rightarrow$ 1.91
        \end{itemize}
      \end{block}
    \end{column}
    \begin{column}{0.5\textwidth}
      \begin{block}{2. Tools: Payload-Dictated}
        \begin{itemize}
          \item C++ metrics independent of step count
          \item Instructions: \textbf{10M} (I/O) $\rightarrow$ \textbf{5B} (DB)
          \item Kernels near $\sim$2.0 IPC (max throughput)
          \item DB scan IPC = 2.03 (prefetcher wins)
          \item AST IPC = 1.99 (branch mispredicts)
        \end{itemize}
      \end{block}
    \end{column}
  \end{columns}
  \vspace{0.3em}
  \begin{center}
    \textcolor{neonpink}{\textbf{LLM coordination overhead is a compounding tax that scales with autonomy.}}
  \end{center}
\end{frame}

\begin{frame}{Optimization 1: The Vectorization Wall (AVX2)}
  \begin{columns}
    \begin{column}{0.55\textwidth}

\end{frame}

\begin{frame}{Limitations}
  \begin{block}{1. PAPI Profiling Artifact (OpenMP)}
    \begin{itemize}
      \item High-level PAPI tracks \textbf{only the calling thread}
      \item Master thread spin-waits on OpenMP barrier $\rightarrow$ cycles accumulate but instructions don't
      \item Reported IPC plummeted to \textbf{0.14} -- a profiling blind spot, not a real regression
      \item \textbf{Fix:} thread-local PAPI EventSets for future multi-threaded work
    \end{itemize}
  \end{block}
  \begin{block}{2. OpenMP Negative Scaling}
    \begin{itemize}
      \item Parallel FS walker used $15\times$ \textit{more} CPU cycles than the serial baseline
      \item Root cause: \texttt{std::atomic} global counter $\rightarrow$ L1 cache-line contention
      \item Plus OS-kernel serialization of small-directory disk reads
    \end{itemize}
  \end{block}
  \begin{block}{3. Static Workloads}
    \begin{itemize}
      \item Sampling disabled; static payloads $\rightarrow$ our numbers are a \textbf{best-case lower bound}
      \item Real LLMs hallucinate $\rightarrow$ multi-turn error correction $\rightarrow$ even worse tax
    \end{itemize}
  \end{block}
\end{frame}

\begin{frame}{Conclusion: Rethinking Agentic Systems}
  \begin{exampleblock}{Headline Findings}
    \begin{itemize}
      \item Lightweight agent tools suffer up to $\mathbf{69.5\times}$ latency slowdown.
      \item Kernel IPC improvements of $\mathbf{14.2\%}$ translate to $\mathbf{<1\%}$ end-to-end speedup.
      \item This is the \textbf{Masking Effect}: framework bloat swallows hardware victories.
    \end{itemize}
  \end{exampleblock}

  \textbf{Future Work:}
  \begin{enumerate}
    \item \textbf{High-throughput inference engines:} continuous batching + PagedAttention (e.g.,\ vLLM).
    \item \textbf{Compiled orchestrators:} migrate Python $\rightarrow$ Rust/C++; use SIMD JSON parsers (\texttt{orjson}, \texttt{simdjson}).
    \item \textbf{Binary protocols:} fine-tune LLMs to emit Apache Arrow instead of JSON strings.
  \end{enumerate}

  \vspace{0.4em}
  \begin{center}
    \textcolor{neonmint}{\textbf{Until the orchestration layer is treated as a high-performance data plane,\\
    the promise of agentic AI will remain masked by its own framework.}}
  \end{center}
\end{frame}
